Implementation Reference
// 1. Load from CDN in <head>
<script src="https://unpkg.com/lucide@latest"></script>

// 2. Use data attribute in HTML
<i data-lucide="wine" class="w-6 h-6 text-wine-400"></i>
<i data-lucide="shopping-bag" class="w-5 h-5"></i>

// 3. Initialize after DOM is ready
document.addEventListener('DOMContentLoaded', () => {
  lucide.createIcons();
});

// 4. Re-initialize after dynamic renders (e.g. cart update)
lucide.createIcons(); // call again after innerHTML changes
